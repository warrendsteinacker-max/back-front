  <button class="btn-red" style="width:100%;margin-top:6px" onclick="clearDB()">🧹 Clear Entire DB</button>
  <pre id="db-snapshot" style="font-size:0.72rem;overflow:auto;max-height:200px;background:#1e1e1e;color:#9cdcfe;padding:10px;border-radius:4px;margin-top:10px"></pre>

<script>
  // ── STATE (mirrors server DB in memory) ──────────────────────────
  let inventory = [];
  let purchases = [];

  let scanner = null;
  let currentBarcode = '';

  // ── UTILITIES ────────────────────────────────────────────────────
  function setStatus(msg, type = 'neutral') {
    const el = document.getElementById('status-bar');
    el.innerText = msg;
    const styles = {
      ok:      { bg: '#d4edda', color: '#155724', border: '#c3e6cb' },
      error:   { bg: '#f8d7da', color: '#721c24', border: '#f5c6cb' },
      loading: { bg: '#fff3cd', color: '#856404', border: '#ffeeba' },
      neutral: { bg: '#e2e3e5', color: '#383d41', border: '#d6d8db' }
    };
    const s = styles[type] || styles.neutral;
    el.style.background = s.bg;
    el.style.color = s.color;
    el.style.border = `1px solid ${s.border}`;
  }

  async function api(method, endpoint, body) {
    const opts = { method, headers: {} };
    if (body) { opts.headers['Content-Type'] = 'application/json'; opts.body = JSON.stringify(body); }
    const res = await fetch(endpoint, opts);
    return res.json();
  }

  // ── PING ─────────────────────────────────────────────────────────
  async function pingTest() {
    setStatus('Pinging server...', 'loading');
    const t = Date.now();
    try {
      const data = await api('GET', '/api/ping');
      setStatus(`✅ Connected! Server responded in ${Date.now()-t}ms.`, 'ok');
      await loadFromServer();
    } catch(e) {
      setStatus('❌ Failed. Check: same Wi-Fi? Cellular data off? Firewall open on port 3000?', 'error');
    }
  }

  // ── LOAD ALL DATA FROM SERVER ────────────────────────────────────
  async function loadFromServer() {
    try {
      const data = await api('GET', '/api/db');
      inventory = data.inventory || [];
      purchases = data.purchases || [];
      document.getElementById('db-snapshot').textContent = JSON.stringify(data, null, 2);
      renderAll();
    } catch(e) {
      console.error('Could not load from server:', e);
    }
  }

  // ── SCANNER ──────────────────────────────────────────────────────
  function toggleScanner() {
    const reader = document.getElementById('reader');
    const btn = document.getElementById('scan-btn');
    if (!scanner) {
      reader.classList.remove('hidden');
      btn.innerText = '⏹ Stop Scanner';
      scanner = new Html5QrcodeScanner('reader', { fps: 10, qrbox: { width: 260, height: 160 } }, false);
      scanner.render(onScanSuccess, () => {});
    } else {
      stopScanner();
    }
  }

  function stopScanner() {
    if (scanner) {
      scanner.clear().catch(() => {});
      scanner = null;
      document.getElementById('reader').classList.add('hidden');
      document.getElementById('scan-btn').innerText = '▶ Start Scanner';
    }
  }

  async function onScanSuccess(code) {
    stopScanner();
    currentBarcode = code;
    document.getElementById('scanned-code').innerText = code;

    // Log raw scan to server
    await api('POST', '/api/scans', { barcode: code });

    // Try to auto-fill name from existing inventory
    const match = inventory.find(i => i.barcode === code);
    document.getElementById('item-name').value = match ? match.name : '';
    document.getElementById('action-box').classList.remove('hidden');
  }

  // ── SAVE ITEM (from scanner action box) ─────────────────────────
  async function saveItem(target) {
    const rawName = document.getElementById('item-name').value.trim();
    const name = rawName || `Item ${currentBarcode}`;
    setStatus('Saving...', 'loading');
    try {
      if (target === 'inventory') {
        const res = await api('POST', '/api/inventory', { barcode: currentBarcode, name });
        setStatus(`✅ "${name}" saved to inventory.`, 'ok');
      } else {
        const res = await api('POST', '/api/purchases', { barcode: currentBarcode, name });
        setStatus(`✅ "${name}" logged as purchase.`, 'ok');
      }
      document.getElementById('action-box').classList.add('hidden');
      document.getElementById('item-name').value = '';
      currentBarcode = '';
      await loadFromServer();
    } catch(e) {
      setStatus('❌ Save failed. Is the server running?', 'error');
    }
  }

  // ── QUICK PURCHASE (from dropdown) ──────────────────────────────
  async function quickPurchase() {
    const sel = document.getElementById('quick-select');
    const id = sel.value;
    if (!id) return;
    const item = inventory.find(i => i.id === id);
    if (!item) return;
    setStatus('Logging purchase...', 'loading');
    try {
      await api('POST', '/api/purchases', { barcode: item.barcode, name: item.name });
      setStatus(`✅ "${item.name}" logged as purchase.`, 'ok');
      await loadFromServer();
    } catch(e) {
      setStatus('❌ Failed to log purchase.', 'error');
    }
  }

  // ── DELETE ───────────────────────────────────────────────────────
  async function deleteItem(id, type) {
    const endpoint = type === 'inventory' ? `/api/inventory/${id}` : `/api/purchases/${id}`;
    try {
      await api('DELETE', endpoint);
      await loadFromServer();
    } catch(e) {
      setStatus('❌ Delete failed.', 'error');
    }
  }

  // ── CLEAR ENTIRE DB ──────────────────────────────────────────────
  async function clearDB() {
    if (!confirm('Clear the entire server database? This cannot be undone.')) return;
    await api('DELETE', '/api/db');
    await loadFromServer();
    setStatus('🧹 Database cleared.', 'neutral');
  }

  // ── RENDER LISTS ─────────────────────────────────────────────────
  function renderAll() {
    const q = document.getElementById('search').value.toLowerCase();

    // Inventory list
    const invEl = document.getElementById('inv-list');
    const filtInv = inventory.filter(i => i.name.toLowerCase().includes(q) || i.barcode.includes(q));
    invEl.innerHTML = filtInv.length === 0
      ? '<li style="color:#888;font-size:0.9rem;padding:5px">No items.</li>'
      : filtInv.map(i => `
          <li class="item-card">
            <div>
              <strong>${i.name}</strong><br>
              <small>Barcode: ${i.barcode} | Added: ${i.createdAt ? new Date(i.createdAt).toLocaleString() : '—'}</small>
            </div>
            <button class="btn-red" onclick="deleteItem('${i.id}','inventory')">Remove</button>
          </li>`).join('');

    // Purchases list
    const purEl = document.getElementById('pur-list');
    const filtPur = purchases.filter(i => i.name.toLowerCase().includes(q) || i.barcode.includes(q));
    purEl.innerHTML = filtPur.length === 0
      ? '<li style="color:#888;font-size:0.9rem;padding:5px">No purchases.</li>'
      : filtPur.map(i => `
          <li class="item-card" style="border-left:4px solid #6f42c1">
            <div>
              <strong>${i.name}</strong><br>
              <small>Logged: ${i.loggedAt ? new Date(i.loggedAt).toLocaleString() : '—'}</small>
            </div>
            <button class="btn-red" onclick="deleteItem('${i.id}','purchase')">Delete</button>
          </li>`).join('');

    // Quick select dropdown
    const sel = document.getElementById('quick-select');
    sel.innerHTML = inventory.length === 0
      ? '<option value="">-- No inventory items --</option>'
      : inventory.map(i => `<option value="${i.id}">${i.name} (${i.barcode})</option>`).join('');
  }

  // ── INIT ─────────────────────────────────────────────────────────
  document.addEventListener('DOMContentLoaded', () => {
    // Try to auto-load if page was served from the Node server
    loadFromServer().catch(() => {});
  });
</script>
</body>
